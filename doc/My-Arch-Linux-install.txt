fdisk -l
fdisk /dev/sda
mkfs.ext4 -L <label> /dev/sdaX
mkswap -L swap /dev/sdaX && swapon -L swap
mount -L arch /mnt
mkdir /mnt/boot && mount -L boot /mnt/boot
mkdir /mnt/home && mount -L home /mnt/home

nano /etc/pacman.d/mirrorlist
wifi-menu
ping -c 3 google.com

pacstrap -i /mnt base base-devel
genfstab -U /mnt >> /mnt/etc/fstab



arch-chroot /mnt /bin/bash
nano /etc/locale.gen
    en_US.UTF-8 UTF-8
locale-gen
echo LANG=en_US.UTF-8 > /etc/locale.conf

tzselect
ln -sf /usr/share/zoneinfo/Asia/Ho_Chi_Minh /etc/localtime
hwclock --systohc --utc
    #timedatectl set-ntp true

pacman -S grub os-prober
grub-install --target=i386-pc --recheck /dev/sda
grub-mkconfig -o /boot/grub/grub.cfg

echo arch > /etc/hostname
nano /etc/hosts
    arch

pacman -S iw wpa_supplicant dialog
passwd
exit
umount -R /mnt
reboot



useradd -m -G wheel -s /bin/bash bvanhai
passwd bvanhai
nano /etc/sudoers
    bvanhai ALL = (All) ALL
exit



wifi-menu
    #ip addr
    #systemctl start dhcpcd@enp0s25.service
    #systemctl enable dhcpcd@enp0s25.service
pacman -S xorg
pacman -S xorg-drivers
pacman -S xorg-xinit
pacman -S bspwm sxhkd
nano ~/.xinitrc
    sxhkd &
    exec bspwm
startx
